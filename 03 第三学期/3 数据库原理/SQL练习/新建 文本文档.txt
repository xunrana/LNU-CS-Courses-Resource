--13.查询操作系统成绩最高的学生[姓名已脱敏]和所在系
select sname,sdept
from student
where sno in
(
select sno from score where cno=(select cno from course where cname='操作系统') and grade=(select max(grade) from score where cno=(select cno from course where cname='操作系统'))
)

--18.查询既选修操作系统课程又选修计算机导论课程的学生[姓名已脱敏]和所在系（要求两种方法）
select sname,sdept
from student
where sno in(select sno from score where cno=(select cno from course where cname='操作系统')) 
and sno in(select sno from score where cno=(select cno from course where cname='计算机导论')) 


select sname, sdept
from student s, score sc, course c
where s.sno=sc.sno and sc.cno=c.cno 
and cname='操作系统'
        and s.sno in ( select sno
                           from score sc, course c
                       where  sc.cno=c.cno and cname='计算机导论' )



--19.查询每门课成绩最高的同学， 要求显示课程名，学生名，所在系，成绩。
select cname,sname,sdept,grade
from student s,score sc,course c
where sc.sno=s.sno and sc.cno=c.cno and grade=(select max(grade) from score sc1 where sc1.cno=sc.cno)
