--1.建表department， 要求部门号为主码， 部门名非空唯一
create table department
(
dno char([[姓名已脱敏]已脱敏]) primary key,
dname varchar(100) unique not null
)
--2.建表teacher， 要求教师号做主码， 部门号为外码，职称包括教授，副教授，讲师和助教， 实发工资不低于3000元。
create table teacher
(
tno char([[姓名已脱敏]已脱敏]) primary key,
tname varchar(100),
title varchar([[姓名已脱敏]已脱敏]) check(title in('教授', '副教授', '讲师', '助教')),
dno char([[姓名已脱敏]已脱敏]),
salary money check(salary>=3000),
deduction money,
foreign key(dno) references department(dno)
)
