--3.将每个顾客的订单数，保存到numofOrders表里。
insert into numoforders
select o.custno,fname,lname,count(*)
from orders o,customer c
where o.custno=c.custno
group by o.custno,fname,lname
select * from numoforders
--4.将每个销售代表的总销售金额保存到totalSale表里。
insert into totalsale
select o.salesrepno,salesreplastname,salesrepfname,sum(qtypurchased*purchaseprice)
from orderline ol,orders o,salesrep s
where ol.orderno=o.orderno and s.salesrepno=o.salesrepno
group by o.salesrepno,salesreplastname,salesrepfname
select * from totalsale
--5.将每个商品单价上调5%。

--6.把购买laser Printer的顾客的余额加100元

(

)

--7.订单数超过2的商品的单价上调5%
update product

where productno in
(
select productno
from orders o,orderline ol
where o.orderno=ol.orderno
group by productno
having count(*)>2
)

--9.创建视图V1，显示顾客和他们购买商品的信息。(要求有个顾客[姓名已脱敏],购买商品日期,商品名,数量,单价,小计(数量*单价)
create view V1 as
select fname,lname,orddate,proddescription,qtypurchased,purchaseprice,qtypurchased*purchaseprice as totalprice
from customer c,orders o,orderline ol,product p
where c.custno=o.custno and o.orderno=ol.orderno and p.productno =ol.productno
select * from V1
--10.创建视图V2，显示销售代表（名）及其所在的部门名
create view V2 as
select salesrepfname,departname
from salesrep s,department d
where s.departno=d.departno
select * from V2
--11.创建视图V3， 显示销售代表([姓名已脱敏])， 部门名和他们的业绩（有多少订单）
create view V3 as
select salesreplastname,salesrepfname,departname,count(*) as sumorders
from salesrep s,department d,orders o
where s.departno=d.departno and s.salesrepno=o.salesrepno
group by salesreplastname,salesrepfname,departname
select * from V3
--12.在V1视图上， 查询每名顾客一共消费的了多少钱， 显示顾客名，总消费金额
select fname,lname,sum(totalprice) as sumcost
from V1
group by fname,lname
--13.在V3视图上，查询订单数超过3个的销售代表[姓名已脱敏]，部门，订单数量
select salesreplastname,salesrepfname,departname,sumorders
from V3
where sumorders>3
--14.删除Tech部门
delete from department
where departname='Tech'
--15.把订单数为零的销售代表消息全部删除
delete from salesrep
where salesrepno in
(
select s.salesrepno
from salesrep s,orders o
where s.salesrepno=o.salesrepno
group by s.salesrepno
having count(*)=0
)
--16.在商品表上按商品名升序建立索引
create index index_product_proddescription on product(proddescription asc)
--17.在销售代表表上按部门号升序建立索引， 如果部门号相同， 按[姓名已脱敏]降序建立索引。
create index index_salesreplastname on salesrep(departno asc,salesreplastname desc,salesrepfname desc)
