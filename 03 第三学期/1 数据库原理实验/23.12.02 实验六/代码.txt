--1. 在数据库retail中， 创建一个名为countSalesRep的存储过程。 该存储过程用于显示（查询）每个部门有多少销售员。
create procedure countsalesrep 
as
begin
select d.departname,count(*) as cnt 
from salesrep s,department d
where s.departno=d.departno
group by d.departname
end
countsalesrep
--2. 在数据库retail中， 创建一个名为 totalOne的存储过程。该存储过程用于显示（查询）指定顾客（顾客号）的消费信息（订单号， 商品名， 购买单价，购买数量， 小计）
create procedure totalone
@custno char(4)
as
begin
select o.orderno,p.proddescription,ol.purchaseprice,ol.qtypurchased,ol.purchaseprice*ol.qtypurchased as total
from orders o,orderline ol,product p
where o.orderno=ol.orderno and p.productno=ol.productno and o.custno=@custno
end
totalone '1040'

--4. 在数据库retail中， 创建一个名为getBalance的函数。 该函数中， 给定一顾客号，返回该顾客账户余额。
create function getbalance(@custno char(4))
returns numeric(8,2)
as 
begin 
     declare @balance numeric(8,2)
	 set @balance=(select balance from customer where custno=@custno)
	 return @balance
end

--6. 在数据库retail中,创建一个名为salesPerformance的函数。该函数可以计算出每位销售员的销售业绩（销售员编号,名,姓,销售业绩）
create function salesperformance()
returns table
as
return
     (
	 select s.salesrepno,salesrepfname,salesreplastname,sum(ol.PurchasePrice*ol.qtypurchased) as salesamount
	 from salesrep s,orders o,orderline ol
	 where o.salesrepno=s.salesrepno and o.orderno=ol.orderno 
	 group by s.salesrepno,salesrepfname,salesreplastname
	 )
--7. 测试salesPerformance函数
select * from salesperformance()
--8. 在数据库retail中,创建一个名为showCustomer的存储过程。在该存储过程里,声明游标,使用该游标一条一条显示顾客信息（顾客号,名,姓,账户余额）。
create procedure showcustomer
as
begin
declare c1 cursor
for select custno,fname,lname,balance from customer
open c1
declare @custno char(4)
declare @fname char([[姓名已脱敏]已脱敏])
declare @lname char([[姓名已脱敏]已脱敏])
declare @balance numeric(8,2)
fetch c1 into @custno,@fname,@lname,@balance
while @@fetch_status=0
begin
     print @custno+' '+@fname+' '+@lname+' '+convert(char,@balance)
	 fetch c1 into @custno,@fname,@lname,@balance
end
close c1
deallocate c1
end
