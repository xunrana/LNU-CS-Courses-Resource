--3.查询选修操作系统学生的名字和成绩

from student s,sc,course c
where s.sno=sc.sno and sc.cno=c.cno and c.cname='操作系统'

--4.查询选修李诚老师所教课程的学生[姓名已脱敏]， [姓名已脱敏]和所在系

from student s,course c,teacher t,sc
where c.tno=t.tno and t.tname='李诚' and sc.cno=c.cno and sc.sno=s.sno

--5.查询选修操作系统成绩超过90分学生的[姓名已脱敏]，所在系和成绩，按成绩降序排列
select sname,sdept,grade
from student s,sc,course c
where c.cname='操作系统'and sc.cno=c.cno and sc.sno=s.sno and grade>90 
order by grade desc

--6.查询没有选修操作系统的学生[姓名已脱敏]和[姓名已脱敏]
select sno,sname

where sno not in(
select sno

--7.查询每名学生已选修的学分， 查询结果要求显示学生[姓名已脱敏]和总学分
select sname,sum(credit) as sumcredit
from student s,course c,sc
where s.sno=sc.sno and c.cno=sc.cno
group by sname,s.sno

--8.查询每门课程已选修的人数， 查询结果要求显示课程名和选课人数
select cname,count(sc.sno) as cnt
from sc,student s,course c
where sc.sno=s.sno and sc.cno=c.cno 
group by c.cno,cname

--9.查询操作系统成绩最高的学生[姓名已脱敏]和所在系

from student s
where sno in (select sno
              from sc,course c
              where sc.cno=c.cno and cname='操作系统' and grade in ( select max(grade) from course c,sc where sc.cno=c.cno and cname='操作系统' )               
              )

--10.查询选课人数超过3人的课程名

from sc,course c
where sc.cno=c.cno
group by c.cno,cname
having count(sc.cno)>3
