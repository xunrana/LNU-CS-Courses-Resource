--4.查询一直都没有售出的商品的商品号,商品名
select p.productno,proddescription
from product p
where not exists (select *

--6.	对于每个订单，显示顾客[姓名已脱敏]，销售代表的[姓名已脱敏]，订单号，订单日期，每个商品名及数量， 按顾客[姓名已脱敏]升序，订单日期降序排列
select lname,fname,salesreplastname,salesrepfname,o.orderno,orddate,proddescription,qtypurchased
from customer c,salesrep s,orders o,orderline ol,product p
where c.custno=o.custno and s.salesrepno=o.salesrepno and o.orderno=ol.orderno and ol.productno=p.productno
order by lname,fname asc,orddate desc

--7.每个订单上一共消费多少钱？ 显示订单号，总计价钱，并将总计价钱重新命名为’Total Amount’
select orderno,sum(qtypurchased*purchaseprice) as 'Tatal Amount'
from orderline 
group by orderno

--8.查询每一个部门里有多少销售代表？ 显示部门名，和销售代表数，并将销售代表数重新命名为’ Number of Employees’
select d.departname,count(s.salesrepno) as 'Number of Employees'
from department d,salesrep s
where d.departno=s.departno
group by d.departname

--9.哪些顾客有超过三个以上的订单，显示顾客[姓名已脱敏]和订单数。
select lname,fname,count(o.orderno) as ordercnt
from customer c,orders o
where o.custno=c.custno 
group by lname,fname
having count(o.orderno) > 3

--10.查询订单状态为’C’和为’N’的订单数， 结果按状态升序排列。
select ordstatus,count(*) as stacnt
from orders
group by ordstatus
order by ordstatus desc

--11.查询受理订单书超过3个的销售代表的[姓名已脱敏]，要求显示销售代表的[姓名已脱敏]，和订单数。
select salesreplastname,salesrepfname,count(*) as ordercnt
from salesrep s,orders o
where s.salesrepno=o.salesrepno
group by salesreplastname,salesrepfname
having count(*)>3

--12.查询订单号为12630的顾客[姓名已脱敏]。
select lname,fname
from customer c,orders o
where o.custno=c.custno and o.orderno=12630

--13.商品号为’DP03’的商品被订了多少次？
select count(*) as ordercnt
from orderline 
where productno='DP03'

--14.查询订购的商品名为’Walkman’的顾客[姓名已脱敏]。
select lname,fname
from customer c,orders o,orderline ol,product p
where c.custno=o.custno and o.orderno=ol.orderno and ol.productno=p.productno and p.proddescription='Walkman'

--15.查询最高的账户余额是多少
select max(balance) as 'Highest Balance'
from customer

--16.查询由销售代表Smith Bill销售过商品的顾客[姓名已脱敏]。
select distinct lname,fname
from customer c,orders o,salesrep s
where c.custno=o.custno and s.salesrepno=o.salesrepno and salesreplastname='Smith' and salesrepfname='Bill'
