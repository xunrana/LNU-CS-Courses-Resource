--2.找出所有艺术家来自中国的歌曲数量统计：
SELECT artists.artistname, COUNT(songs.songid) AS song_count
FROM artists, songs, countries
WHERE artists.artistid = songs.artistid
AND artists.countryid = countries.countryid
AND countries.countryname = '中国'
GROUP BY artists.artistname;
