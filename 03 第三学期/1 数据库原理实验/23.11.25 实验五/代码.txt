-- Teacher(TeacherNO, TeacherName, age, Gender, title ,salary, departNO)

create table teacher 
(

--4.在teacher表里添加约束,职称取值为教授,副教授,讲师,助教。
alter table teacher
add constraint c1 check(title in('教授','副教授','讲师','助教')) 
--5.在teacher表里添加约束,职称为教授的老师, 工资应超过6000
alter table teacher

--8.在数据库sy5里, 定义触发器当顾客的信用卡额度提高超过10%以上时,将相关信息复制到unusual这个表里。
create trigger c_t1
on customer
after update 
as 
begin 
insert into unusual
select d.custno,d.fname,d.lname,d.creditlimit,i.creditlimit
from deleted d,inserted i
where d.custno=i.custno and i.creditlimit>d.creditlimit*1.1
end

--10.在数据库sy5里, 定义触发器当一个销售代表离开公司了,将该销售代表从表SalesRep中删除,同时将该销售代表的信息拷贝到FormerSalesRep表里。

on salesrep

--11.在数据库sy5里,定义触发器,当在orderline表里添加新的数据时,自动更新客户的账户余额和商品的库存量
create trigger c_t3
on orderline
after insert
as
begin
update customer
set balance=balance-(select qtypurchased*purchaseprice from inserted)
update product
set qtyonhand=qtyonhand-(select qtypurchased from inserted)
end
