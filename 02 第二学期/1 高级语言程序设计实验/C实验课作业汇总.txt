2、输入三边长，输出三角形面积。如果三边不能构成三角形，则输出-1。
#include <stdio.h>
#include <math.h>
void main()
{
	double a,b,c,p,s;
	scanf("%lf %lf %lf",&a,&b,&c);
	p=(a+b+c)/2;
	if(a+b>c&&a+c>b&&b+c>a)
	{
		s=sqrt(p*(p-a)*(p-b)*(p-c));
		printf("三角形三边长%.2f,%.2f,%.2f 面积为%.4f\n",a,b,c,s);
	}
	else printf("-1\n");
}

{

	p=b*b-4*a*c;
	if(p>=0)
	{
		x1=(-b+sqrt(p))/(2*a);
		x2=(-b-sqrt(p))/(2*a);
		printf("一元二次方程的解为：x1=%.3f   x2=%.3f\n",x1,x2);
	}
	else printf("方程无实根\n");
}

10、编程：将从键盘上输入的整数转为逆序的整数。

{
	int n,m,k;
	scanf("%d",&n);
	m=n;
	k=0;
	while(m!=0)
	{
		k=k*10+m%10;
		m=m/10;
	}

void arr_sort(int a[],int n)
{
	int i,j,temp,flag;
	for(i=0;i<n-1;i++)
	{
		flag=0;
		for(j=0;j<n-1-i;j++)
			if(a[j]<a[j+1])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
				flag=1;
			}		
		if(flag==0)break;
	}
}
